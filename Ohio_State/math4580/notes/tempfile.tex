\begin{fact}
    \begin{enumerate}
        Distribution properties hold.
        \item \(A \cdot (B + C) = A \cdot B + A \cdot C\)
        \item \((A + B) \cdot C = A \cdot C + B \cdot C\)
    \end{enumerate}
\end{fact}
\begin{proof}
    We will compute the left hand side with the right hand side and show they are equal.
    \begin{align*}
        v & = A \cdot (B + C)(v)          \\
          & = A \cdot (B + C)(v)          \\
          & = A (B(v) + C(v))             \\
          & = A(B(v)) + A(C(v))           \\
          & = A \cdot B(v) + A \cdot C(v) \\
          & = A \cdot B + A \cdot C(v)
    \end{align*}
\end{proof}
